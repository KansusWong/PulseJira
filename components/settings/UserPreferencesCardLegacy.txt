"LEGACY UserPreferencesCard archived after migration to UserPreferencesCardV2";

import { useEffect, useState, useCallback } from "react";
import {
  Target,
  Plus,
  X,
  Save,
  Loader2,
  CheckCircle2,
  AlertCircle,
  Info,
  Zap,
} from "lucide-react";
import clsx from "clsx";

interface PlatformConfig {
  enabled: boolean;
  sources: string[];
}

interface UserPreferences {
  topics: string[];
  platforms: {
    reddit: PlatformConfig;
    twitter: PlatformConfig;
    youtube: PlatformConfig;
  };
  updatedAt: string;
}

interface AvailablePlatforms {
  reddit: boolean;
  twitter: boolean;
  youtube: boolean;
}

type PlatformKey = "reddit" | "twitter" | "youtube";

const platformMeta: Record<
  PlatformKey,
  { icon: string; label: string; color: string; sourcePlaceholder: string; sourceLabel: string }
> = {
  reddit: {
    icon: "\u{1F4AC}",
    label: "Reddit",
    color: "orange",
    sourcePlaceholder: "e.g. reactjs, programming, webdev",
    sourceLabel: "Subreddits（可选覆盖）",
  },
  twitter: {
    icon: "\u{1F426}",
    label: "Twitter",
    color: "sky",
    sourcePlaceholder: "e.g. AI coding, LLM tools, developer productivity",
    sourceLabel: "搜索关键词（可选覆盖）",
  },
  youtube: {
    icon: "\u{1F4F9}",
    label: "YouTube",
    color: "red",
    sourcePlaceholder: "e.g. coding assistant, AI dev tools",
    sourceLabel: "搜索关键词（可选覆盖）",
  },
};

const colorMap: Record<string, { bg: string; border: string; text: string; dot: string }> = {
  orange: { bg: "bg-orange-500/10", border: "border-orange-500/20", text: "text-orange-400", dot: "bg-orange-500" },
  sky: { bg: "bg-sky-500/10", border: "border-sky-500/20", text: "text-sky-400", dot: "bg-sky-500" },
  red: { bg: "bg-red-500/10", border: "border-red-500/20", text: "text-red-400", dot: "bg-red-500" },
};

type SaveStatus = "idle" | "saving" | "saved" | "error";

export function UserPreferencesCardLegacy() {
  const [loading, setLoading] = useState(true);
  const [preferences, setPreferences] = useState<UserPreferences>({
    topics: [],
    platforms: {
      reddit: { enabled: false, sources: [] },
      twitter: { enabled: false, sources: [] },
      youtube: { enabled: false, sources: [] },
    },
    updatedAt: "",
  });
  const [available, setAvailable] = useState<AvailablePlatforms>({
    reddit: false,
    twitter: false,
    youtube: false,
  });
  const [topicInput, setTopicInput] = useState("");
  const [sourceInputs, setSourceInputs] = useState<Record<PlatformKey, string>>({
    reddit: "",
    twitter: "",
    youtube: "",
  });
  const [saveStatus, setSaveStatus] = useState<SaveStatus>("idle");

  useEffect(() => {
    fetch("/api/settings/preferences")
      .then((res) => res.json())
      .then((json) => {
        if (json.success && json.data) {
          setPreferences(json.data.preferences);
          setAvailable(json.data.availablePlatforms);
        }
      })
      .catch(console.error)
      .finally(() => setLoading(false));
  }, []);

  const handleAddTopic = useCallback(() => {
    const topic = topicInput.trim();
    if (!topic) return;
    if (preferences.topics.includes(topic)) {
      setTopicInput("");
      return;
    }
    setPreferences((prev) => ({ ...prev, topics: [...prev.topics, topic] }));
    setTopicInput("");
  }, [topicInput, preferences.topics]);

  const handleRemoveTopic = useCallback((topic: string) => {
    setPreferences((prev) => ({
      ...prev,
      topics: prev.topics.filter((t) => t !== topic),
    }));
  }, []);

  const handleAddSource = useCallback(
    (platform: PlatformKey) => {
      const value = sourceInputs[platform].trim();
      if (!value) return;
      const items = value
        .split(",")
        .map((s) => s.trim())
        .filter((s) => s && !preferences.platforms[platform].sources.includes(s));
      if (items.length === 0) {
        setSourceInputs((prev) => ({ ...prev, [platform]: "" }));
        return;
      }
      setPreferences((prev) => ({
        ...prev,
        platforms: {
          ...prev.platforms,
          [platform]: {
            ...prev.platforms[platform],
            sources: [...prev.platforms[platform].sources, ...items],
          },
        },
      }));
      setSourceInputs((prev) => ({ ...prev, [platform]: "" }));
    },
    [sourceInputs, preferences.platforms]
  );

  const handleRemoveSource = useCallback((platform: PlatformKey, source: string) => {
    setPreferences((prev) => ({
      ...prev,
      platforms: {
        ...prev.platforms,
        [platform]: {
          ...prev.platforms[platform],
          sources: prev.platforms[platform].sources.filter((s) => s !== source),
        },
      },
    }));
  }, []);

  const handleSave = async () => {
    setSaveStatus("saving");
    try {
      const res = await fetch("/api/settings/preferences", {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          topics: preferences.topics,
          platforms: preferences.platforms,
        }),
      });
      const json = await res.json();
      if (json.success) {
        setSaveStatus("saved");
        setTimeout(() => setSaveStatus("idle"), 2000);
      } else {
        setSaveStatus("error");
      }
    } catch {
      setSaveStatus("error");
    }
  };

  const hasTopics = preferences.topics.length > 0;
  const availableCount = Object.values(available).filter(Boolean).length;

  if (loading) {
    return (
      <div className="bg-paper border border-border rounded-xl p-6 flex items-center justify-center min-h-[200px]">
        <Loader2 className="w-5 h-5 animate-spin text-zinc-500" />
      </div>
    );
  }

  return (
    <div className="bg-paper border border-border rounded-xl overflow-hidden">
      {/* Header */}
      <div className="px-6 py-5 border-b border-border">
        <div className="flex items-center gap-3">
          <div className="p-2 bg-amber-500/10 rounded-lg">
            <Target className="w-5 h-5 text-amber-400" />
          </div>
          <div>
            <h2 className="text-base font-bold text-zinc-100">
              信号偏好
            </h2>
            <p className="text-xs text-zinc-500 mt-0.5">
              设定感兴趣的话题，系统将自动从已连接的平台抓取相关信号
            </p>
          </div>
        </div>
      </div>

      <div className="p-6 space-y-6">
        {/* Topics Section */}
        <div>
          <label className="text-sm font-medium text-zinc-300 mb-3 block">
            感兴趣的话题
          </label>

          {/* Tags */}
          {preferences.topics.length > 0 && (
            <div className="flex flex-wrap gap-2 mb-3">
              {preferences.topics.map((topic) => (
                <span
                  key={topic}
                  className="inline-flex items-center gap-1.5 px-3 py-1.5 text-xs font-medium bg-zinc-800 text-zinc-300 rounded-full border border-zinc-700"
                >
                  {topic}
                  <button
                    onClick={() => handleRemoveTopic(topic)}
                    className="text-zinc-500 hover:text-zinc-200 transition-colors"
                  >
                    <X className="w-3 h-3" />
                  </button>
                </span>
              ))}
            </div>
          )}

          {/* Add topic input */}
          <div className="flex gap-2">
            <input
              type="text"
              value={topicInput}
              onChange={(e) => setTopicInput(e.target.value)}
              onKeyDown={(e) => {
                if (e.key === "Enter") {
                  e.preventDefault();
                  handleAddTopic();
                }
              }}
              placeholder="输入话题关键词，按 Enter 添加..."
              className="flex-1 bg-zinc-900 border border-zinc-700 rounded-lg px-3 py-2 text-sm text-zinc-300 placeholder:text-zinc-600 focus:outline-none focus:border-zinc-500 transition-colors"
            />
            <button
              onClick={handleAddTopic}
              disabled={!topicInput.trim()}
              className="flex items-center gap-1.5 px-3 py-2 text-xs font-medium bg-zinc-800 hover:bg-zinc-700 text-zinc-300 rounded-lg border border-zinc-700 transition-colors disabled:opacity-30 disabled:cursor-not-allowed"
            >
              <Plus className="w-3.5 h-3.5" />
              添加
            </button>
          </div>

          {preferences.topics.length === 0 && (
            <p className="text-[11px] text-zinc-600 mt-2 flex items-center gap-1.5">
              <Info className="w-3 h-3" />
              例如：AI coding tools, developer productivity, React, multi-agent
            </p>
          )}
        </div>

        {/* Auto-enable hint */}
        {hasTopics && availableCount > 0 && (
          <div className="flex items-center gap-2 px-3 py-2 rounded-lg bg-emerald-500/5 border border-emerald-500/10">
            <Zap className="w-3.5 h-3.5 text-emerald-400 shrink-0" />
            <p className="text-[11px] text-emerald-400/80">
              已连接 {availableCount} 个平台，保存后将自动使用话题关键词采集信号
            </p>
          </div>
        )}

        {/* Platforms Section */}
        <div>
          <label className="text-sm font-medium text-zinc-300 mb-3 block">
            信号来源平台
          </label>

          <div className="space-y-3">
            {(["twitter", "youtube", "reddit"] as PlatformKey[]).map(
              (platform) => {
                const meta = platformMeta[platform];
                const colors = colorMap[meta.color];
                const config = preferences.platforms[platform];
                const isAvailable = available[platform];

                return (
                  <div
                    key={platform}
                    className={clsx(
                      "rounded-lg border p-4 transition-colors",
                      isAvailable
                        ? `${colors.bg} ${colors.border}`
                        : "bg-zinc-900/50 border-zinc-800"
                    )}
                  >
                    {/* Platform header */}
                    <div className="flex items-center justify-between">
                      <div className="flex items-center gap-2.5">
                        <span className="text-base">{meta.icon}</span>
                        <span className="text-sm font-medium text-zinc-200">
                          {meta.label}
                        </span>
                        {!isAvailable ? (
                          <span className="text-[10px] font-mono px-2 py-0.5 rounded-full bg-zinc-800 text-zinc-500 border border-zinc-700">
                            未配置 API Key
                          </span>
                        ) : (
                          <span
                            className={clsx(
                              "text-[10px] font-mono px-2 py-0.5 rounded-full flex items-center gap-1",
                              colors.bg,
                              colors.text
                            )}
                          >
                            <span className={clsx("w-1.5 h-1.5 rounded-full", colors.dot)} />
                            自动采集中
                          </span>
                        )}
                      </div>
                    </div>

                    {/* Sources (visible when platform is available) */}
                    {isAvailable && (
                      <div className="mt-3 pl-7">
                        <label className="text-[11px] text-zinc-500 font-medium mb-2 block uppercase tracking-wider">
                          {meta.sourceLabel}
                        </label>

                        {/* Existing source tags */}
                        {config.sources.length > 0 && (
                          <div className="flex flex-wrap gap-1.5 mb-2">
                            {config.sources.map((source) => (
                              <span
                                key={source}
                                className={clsx(
                                  "inline-flex items-center gap-1 px-2 py-1 text-[11px] rounded-md border",
                                  colors.bg,
                                  colors.border,
                                  colors.text
                                )}
                              >
                                {source}
                                <button
                                  onClick={() =>
                                    handleRemoveSource(platform, source)
                                  }
                                  className="opacity-60 hover:opacity-100 transition-opacity"
                                >
                                  <X className="w-2.5 h-2.5" />
                                </button>
                              </span>
                            ))}
                          </div>
                        )}

                        {/* Add source input */}
                        <div className="flex gap-2">
                          <input
                            type="text"
                            value={sourceInputs[platform]}
                            onChange={(e) =>
                              setSourceInputs((prev) => ({
                                ...prev,
                                [platform]: e.target.value,
                              }))
                            }
                            onKeyDown={(e) => {
                              if (e.key === "Enter") {
                                e.preventDefault();
                                handleAddSource(platform);
                              }
                            }}
                            placeholder={meta.sourcePlaceholder}
                            className="flex-1 bg-black/30 border border-zinc-700/50 rounded-md px-2.5 py-1.5 text-xs text-zinc-300 placeholder:text-zinc-600 focus:outline-none focus:border-zinc-500 transition-colors"
                          />
                          <button
                            onClick={() => handleAddSource(platform)}
                            disabled={!sourceInputs[platform].trim()}
                            className="px-2 py-1.5 text-[11px] font-medium bg-zinc-800/80 hover:bg-zinc-700 text-zinc-400 rounded-md transition-colors disabled:opacity-30 disabled:cursor-not-allowed"
                          >
                            <Plus className="w-3 h-3" />
                          </button>
                        </div>

                        {config.sources.length === 0 && (
                          <p className="text-[10px] text-zinc-600 mt-1.5">
                            不填则默认使用上方话题关键词搜索
                          </p>
                        )}
                      </div>
                    )}
                  </div>
                );
              }
            )}
          </div>
        </div>

        {/* Save button */}
        <div className="flex items-center justify-between pt-2">
          <div className="text-[11px] text-zinc-600">
            {preferences.updatedAt && preferences.topics.length > 0 && (
              <>
                上次更新：
                {new Date(preferences.updatedAt).toLocaleString("zh-CN")}
              </>
            )}
          </div>
          <button
            onClick={handleSave}
            disabled={saveStatus === "saving"}
            className={clsx(
              "flex items-center gap-2 px-4 py-2 text-sm font-medium rounded-lg transition-all",
              saveStatus === "saved"
                ? "bg-emerald-500/20 text-emerald-400"
                : saveStatus === "error"
                ? "bg-red-500/20 text-red-400"
                : "bg-zinc-100 text-zinc-900 hover:bg-white"
            )}
          >
            {saveStatus === "saving" && (
              <Loader2 className="w-3.5 h-3.5 animate-spin" />
            )}
            {saveStatus === "saved" && (
              <CheckCircle2 className="w-3.5 h-3.5" />
            )}
            {saveStatus === "error" && (
              <AlertCircle className="w-3.5 h-3.5" />
            )}
            {saveStatus === "idle" && <Save className="w-3.5 h-3.5" />}
            {saveStatus === "saving"
              ? "保存中..."
              : saveStatus === "saved"
              ? "已保存"
              : saveStatus === "error"
              ? "保存失败"
              : "保存偏好"}
          </button>
        </div>
      </div>
    </div>
  );
}
